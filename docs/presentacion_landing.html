    uint16_t registers[NUM_REGISTERS];  // R0-R15
    uint16_t pc;                   // Program Counter
    uint16_t ir;                   // Instruction Register
    bool halted;
    uint32_t cycle_count;

void cpu_cycle(CPU *cpu) {
    // FETCH
    cpu->ir = cpu->memory[cpu->pc];
    
    // DECODE
    uint8_t opcode = (cpu->ir >> 12) & 0xF;
    uint8_t rd = (cpu->ir >> 8) & 0xF;
    uint8_t operand = cpu->ir & 0xFF;
    
    // EXECUTE
    switch(opcode) {
        case OP_MOVI:  // R[rd] = operand
            cpu->registers[rd] = operand;
            break;
        case OP_SUMAR: // R[rd] += R[operand]
            cpu->registers[rd] += cpu->registers[operand];
            break;
        // ... m√°s instrucciones
    }
    
    cpu->pc++;
